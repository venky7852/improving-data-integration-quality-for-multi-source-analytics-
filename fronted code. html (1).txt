    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
    <script>
        function handleFiles() {
            const input = document.getElementById('fileInput');
            const files = input.files;
            if (files.length === 0) {
                alert('Please upload at least one dataset.');
                return;
            }
            alert(`${files.length} file(s) uploaded successfully!`);
            visualizeDataQuality();
        }

        function visualizeDataQuality() {
            const ctx = document.getElementById('dataChart').getContext('2d');
            new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['Completeness', 'Accuracy', 'Consistency', 'Timeliness'],
                    datasets: [{
                        label: 'Data Quality Metrics',
                        data: [80, 90, 75, 85], // Example data values
                        backgroundColor: ['red', 'blue', 'green', 'orange']
                    }]
                }
            });
        }
    </script>
